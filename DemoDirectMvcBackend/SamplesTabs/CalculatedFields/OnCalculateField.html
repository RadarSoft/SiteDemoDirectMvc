private void Cube_OnCalculateField(object sender, TCalculateFieldArgs e) { // Setting values of the "Sales" calculated measure. if (e.ThisTable("Order Details")) { e.Row["Sales"] = Convert.ToDouble(e.Row["Quantity"]) * Convert.ToDouble(e.Row["UnitPrice"]) * (1 - Convert.ToDouble(e.Row["Discount"])); } // Setting members of the "Employee Name" calculated hierarchy. if (e.ThisTable("Employees")) { e.Row["Employee Name"] = e.Row["FirstName"] + " " + e.Row["LastName"]; } }
